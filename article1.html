-=常用命令行指令=-

pwd //显示当前目录路径

cd //切换目录

ls //显示当前目录下的文件

cd .. //返回上一级目录

-=Git命令=-

初始化-init

git version //查看Git版本

git config --global user.name //设置用户名

git config --global user.email //设置邮箱

git init //初始化仓库

提交-commit

git add 文件吗 //添加文件

git add . //添加所有文件

git commit //提交(打开vim编辑器输入提交说明)

-vim中按a或i进入编辑模式 输入说明-

-编辑模式中按esc退出编辑模式，输入:wq保存并退出-

git commit -m "提交说明" //提交并添加说明(简化命令)

git log //查看提交记录

-vs code 有提交面板-

git reset --hard (id) //回退到指定版本（hard模式 丢弃工作区改动）

-有hard模式和soft模式和mixed模式-

分支-branch

git branch (name) //创建分支

git checkout (name) //切换分支

git merge (name) //合并分支

git branch -d (name) //删除分支

git branch -D (name) //强制删除分支

github协同开发

commit后git branch -M main /创建main分支，并把主分支切换为main

git remote add origin (url) //关联远程仓库(之后输入邮箱、密码)

git clone (url) . //克隆远程仓库

git remote -v //查看远程仓库

git remote remove (remote-name) //删除远程仓库

git remote add upstream (url) //关联上游仓库

git branch -b (name) //创建并切换分支

-创建一个文件夹-

-创建一个json文件-

-写入{ name: "name", url: "url" }-

过一遍add commit后push

git push -u origin (name) //推送分支到远程仓库(末尾加 --force 强制提交)

-进入上游github进入pr提交-

git merge (name)/main //合并分支到主分支
